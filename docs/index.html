Quick Start

# 1. Create config.yaml
cat > config.yaml << EOF
database:
host: your-db-host
username: your-username
password: your-password
backup:
databases: [production_app, user_data]
EOF
# 2. Run backup
docker run -v $(pwd)/config.yaml:/config.yaml ghcr.io/abdullahainun/tenangdb:latest backup
